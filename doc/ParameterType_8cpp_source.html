151 // Generic sign extension
152 template <typename type>
153 void CParameterType::doSignExtend(type& data) const
154 {
155  size_t shift = CHAR_BIT * (sizeof(data) - getSize());
156  // FIXME: If `data` has a signed type and nonnegative value,
157  // and `data × 2^shift` is representable in the result type,
158  // then that is the resulting value;
159  // otherwise, **the behavior is undeﬁned**.
160  // ISO C99 (6.5.7/4) & ISO C++11 [expr.shift]
161  data = (data << shift) >> shift;
162 }

176 // Generic encodability check
177 template <typename type>
178 bool CParameterType::doIsEncodable(type data, bool bIsSigned) const
179 {
180  if (getSize() == sizeof(data)) {
181  // Prevent inappropriate shifts
182  return true;
183  }
184 
185  size_t shift = getSize() * 8;
186 
187  if (!bIsSigned) {
188 
189  // Check high bits are clean
190  return !(data >> shift);
191 
192  } else {
193 
194  // Negative value?
195  bool bIsValueExpectedNegative = (data & (type(1) << (shift - 1))) != 0;
196 
197  // Check high bits are clean
198  return bIsValueExpectedNegative ? !(~data >> shift) : !(data >> shift);
199  }
200 }
