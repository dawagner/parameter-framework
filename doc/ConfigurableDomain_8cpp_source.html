254 bool CConfigurableDomain::parseConfigurableElements(const CXmlElement& xmlElement,
255  CXmlDomainImportContext& serializingContext)
256 {
257  CSystemClass& systemClass = serializingContext.getSystemClass();
258 
259  // Get ConfigurableElements element
260  CXmlElement xmlConfigurableElementsElement;
261  xmlElement.getChildElement("ConfigurableElements", xmlConfigurableElementsElement);
262 
263  // Parse it and associate found configurable elements to it
264  CXmlElement::CChildIterator it(xmlConfigurableElementsElement);
265 
266  CXmlElement xmlConfigurableElementElement;
267 
268  while (it.next(xmlConfigurableElementElement)) {
269 
270  // Locate configurable element
271  string strConfigurableElementPath;
272  xmlConfigurableElementElement.getAttribute("Path", strConfigurableElementPath);
273 
274  CPathNavigator pathNavigator(strConfigurableElementPath);
275  string strError;
276 
277  // Is there an element and does it match system class name?
278  if (!pathNavigator.navigateThrough(systemClass.getName(), strError)) {
279 
280  serializingContext.setError("Could not find configurable element of path " + strConfigurableElementPath + " from ConfigurableDomain description " + getName() + " (" + strError + ")");
281 
282  return false;
283  }
284  // Browse system class for configurable element
285  CConfigurableElement* pConfigurableElement =
286  static_cast<CConfigurableElement*>(systemClass.findDescendant(pathNavigator));
287 
288  if (!pConfigurableElement) {
289 
290  serializingContext.setError("Could not find configurable element of path " + strConfigurableElementPath + " from ConfigurableDomain description " + getName());
291 
292  return false;
293  }
294  // Add found element to domain
295  core::Results infos;
296  if (!addConfigurableElement(pConfigurableElement, NULL, infos)) {
297 
298  strError = utility::asString(infos);
299  serializingContext.setError(strError);
300 
301  return false;
302  }
303  }
304 
305  return true;
306 }

458 // Domain splitting
459 bool CConfigurableDomain::split(CConfigurableElement* pConfigurableElement,
460  core::Results& infos)
461 {
462  // Not associated?
463  if (!containsConfigurableElement(pConfigurableElement)) {
464 
465  std::string strError = "Configurable element " + pConfigurableElement->getPath()
466  + " not associated to configuration domain " + getName();
467  infos.push_back(strError);
468 
469  return false;
470  }
471 
472  // Create sub domain areas for all configurable element's children
473  size_t uiNbConfigurableElementChildren = pConfigurableElement->getNbChildren();
474 
475  if (!uiNbConfigurableElementChildren) {
476 
477  std::string strError = "Configurable element " + pConfigurableElement->getPath()
478  + " has no children to split configurable domain to";
479  infos.push_back(strError);
480 
481  return false;
482  }
483 
484  for (size_t uiChild = 0; uiChild < uiNbConfigurableElementChildren; uiChild++) {
485 
486  CConfigurableElement* pChildConfigurableElement = static_cast<CConfigurableElement*>(pConfigurableElement->getChild(uiChild));
487 
488  doAddConfigurableElement(pChildConfigurableElement, infos);
489  }
490 
491  // Delegate to configurations
492  size_t uiNbConfigurations = getNbChildren();
493 
494  for (size_t uiChild = 0; uiChild < uiNbConfigurations; uiChild++) {
495 
496  CDomainConfiguration* pDomainConfiguration = static_cast<CDomainConfiguration*>(getChild(uiChild));
497 
498  pDomainConfiguration->split(pConfigurableElement);
499  }
500 
501  // Remove given configurable element from this domain
502  // Note: we shouldn't need to recompute the sync set in that case, as the splitted element should include the syncers of its children elements
503  doRemoveConfigurableElement(pConfigurableElement, false);
504 
505  return true;
506 }

525 // Configuration application if required
526 void CConfigurableDomain::apply(CParameterBlackboard* pParameterBlackboard,
527  CSyncerSet* pSyncerSet,
528  bool bForce,
529  std::string& strInfo) const
530 {
531  // Apply configuration only if the blackboard will
532  // be synchronized either now or by syncerSet.
533  if(!pSyncerSet ^ _bSequenceAware) {
534  // The configuration can not be syncronised
535  return;
536  }
537 
538  if (bForce) {
539  // Force a configuration restore by forgetting about last applied configuration
540  _pLastAppliedConfiguration = NULL;
541  }
542  const CDomainConfiguration* pApplicableDomainConfiguration = findApplicableDomainConfiguration();
543 
544  if (pApplicableDomainConfiguration) {
545 
546  // Check not the last one before applying
547  if (!_pLastAppliedConfiguration || _pLastAppliedConfiguration != pApplicableDomainConfiguration) {
548 
549  strInfo = "Applying configuration '" + pApplicableDomainConfiguration->getName()
550  + "' from domain '" + getName() + "'";
551 
552  // Check if we need to synchronize during restore
553  bool bSync = !pSyncerSet && _bSequenceAware;
554 
555  // Do the restore
556  pApplicableDomainConfiguration->restore(pParameterBlackboard, bSync, NULL);
557 
558  // Record last applied configuration
559  _pLastAppliedConfiguration = pApplicableDomainConfiguration;
560 
561  // Check we need to provide syncer set to caller
562  if (pSyncerSet && !_bSequenceAware) {
563 
564  // Since we applied changes, add our own sync set to the given one
565  *pSyncerSet += _syncerSet;
566  }
567  }
568  }
569 }

984 void CConfigurableDomain::mergeAlreadyAssociatedDescendantConfigurableElements(
985  CConfigurableElement* newElement,
986  core::Results& infos)
987 {
988  std::list<CConfigurableElement*> mergedConfigurableElementList;
989 
990  ConfigurableElementListIterator it;
991 
992  // Browse all configurable elements (new one not yet in the list!)
993  for (it = _configurableElementList.begin(); it != _configurableElementList.end(); ++it) {

1006

1014 
1015  CConfigurableElement* pMergedConfigurableElement = *it;

1019  doRemoveConfigurableElement(pMergedConfigurableElement, false);
