130  // Get

135

253 bool CArrayParameter::setValues(size_t uiStartIndex, size_t baseOffset, const string& strValue, CParameterAccessContext& parameterAccessContext) const
254 {
255  // Deal with value(s)
256  Tokenizer tok(strValue, Tokenizer::defaultDelimiters + ",");
257 
258  std::vector<string> astrValues = tok.split();
259  size_t nbValues = astrValues.size();
260 
261  // Check number of provided values
262  if (nbValues + uiStartIndex > getArrayLength()) {
263 
264  // Out of bounds
265  parameterAccessContext.setError("Too many values provided");
266 
267  return false;
268  }
269 
270  // Process
271  size_t valueIndex;
272  size_t size = getSize();
273  size_t offset = getOffset() + uiStartIndex * size - baseOffset;
274 
275  for (valueIndex = 0; valueIndex < nbValues; valueIndex++) {
276 
277  if (!doSetValue(astrValues[valueIndex], offset, parameterAccessContext)) {
278

282 
283  return false;
284  }
285 
286  offset += size;
287  }
288  return true;
