165 bool PfwHandler::createCriteria(const PfwCriterion criteriaArray[], size_t criterionNb)
166 {
167  Status &status = lastStatus;
168  // Add criteria
169  for (size_t criterionIndex = 0; criterionIndex < criterionNb; ++criterionIndex) {
170  const PfwCriterion &criterion = criteriaArray[criterionIndex];
171  if (criterion.name == NULL) {
172  return status.failure("Criterion name is NULL");
173  }
174  if (criterion.values == NULL) {
175  return status.failure("Criterion values is NULL");
176  }
177  // Check that the criterion does not exist
178  if (criteria.find(criterion.name) != criteria.end()) {
179  return status.failure("Criterion \"" + string(criterion.name) +
180  "\" already exist");
181  }
182 
183  // Create criterion type
184  ISelectionCriterionTypeInterface *type =
185  pfw->createSelectionCriterionType(criterion.inclusive);
186  assert(type != NULL);
187  // Add criterion values
188  for (size_t valueIndex = 0; criterion.values[valueIndex] != NULL; ++valueIndex) {
189  int value;
190  if (criterion.inclusive) {
191  // Check that (int)1 << valueIndex would not overflow (UB)
192  if(std::numeric_limits<int>::max() >> valueIndex == 0) {
193  return status.failure("Too many values for criterion " +
194  string(criterion.name));
195  }
196  value = 1 << valueIndex;
197  } else {
198  value = static_cast<int>(valueIndex);
199  }
200  const char * valueName = criterion.values[valueIndex];
201  string error;
202  if (not type->addValuePair(value, valueName, error)) {
203  return status.failure("Could not add value " + string(valueName) +
204  " to criterion " + criterion.name + ": " + error);
205  }
206  }
207  // Create criterion and add it to the pfw
208  criteria[criterion.name] = pfw->createSelectionCriterion(criterion.name, type);
209  }
210  return status.success();
211 }

214 bool pfwStart(PfwHandler *handle, const char *configPath,
215  const PfwCriterion criteria[], size_t criterionNb,
216  const PfwLogger *logger)
217 {
218  // Check that the api is correctly used
219  Status &status = handle->lastStatus;
220 
221  if (handle->pfw != NULL) {
222  return status.failure("Can not start an already started parameter framework");
223  }
224  // Create a pfw
225  handle->pfw = new CParameterMgrPlatformConnector(configPath);
226 
227  handle->setLogger(logger);
228 
229  if (not handle->createCriteria(criteria, criterionNb)) {
230  return status.failure();
