98 {

103  setDescription(description);
104 
105  // Context
106  CXmlParameterSerializingContext& parameterBuildContext = static_cast<CXmlParameterSerializingContext&>(serializingContext);
107 
108  // Install temporary component library for further component creation
109  parameterBuildContext.setComponentLibrary(_pComponentLibrary);
110 
111  CXmlElement childElement;
112 
113  // Manage mapping attribute
114  string rawMapping;
115  xmlElement.getAttribute("Mapping", rawMapping);
116  if (!rawMapping.empty()) {
117 
118  std::string error;
119  _pMappingData = new CMappingData;
120  if (!_pMappingData->init(rawMapping, error)) {
121 
122  serializingContext.setError("Invalid Mapping data from XML element '" +

126  }
127

129  xmlElement.getChildElement("ComponentLibrary", childElement);
130

145

159 {
160  // Default mapping context

362 // Subsystem object creation handling
363 bool CSubsystem::handleSubsystemObjectCreation(
364  CInstanceConfigurableElement* pInstanceConfigurableElement,
365  CMappingContext& context, bool& bHasCreatedSubsystemObject, string& strError)
366 {
367  bHasCreatedSubsystemObject = false;
368 
369  for (const auto *pSubsystemObjectCreator : _subsystemObjectCreatorArray) {
370 
371  // Mapping key
372  string strKey = pSubsystemObjectCreator->getMappingKey();
373  // Object id
374  const string* pStrValue;
375 
376  if (pInstanceConfigurableElement->getMappingData(strKey, pStrValue)) {
377 
378  // First check context consistency
379  // (required ancestors must have been set prior to object creation)
380  uint32_t uiAncestorMask = pSubsystemObjectCreator->getAncestorMask();
381 
382  for (size_t uiAncestorKey = 0; uiAncestorKey < _contextMappingKeyArray.size(); uiAncestorKey++) {
383 
384  if (!((1 << uiAncestorKey) & uiAncestorMask)) {
385  // Ancestor not required
386  continue;
387  }
388  // Check ancestor was provided
389  if (!context.iSet(uiAncestorKey)) {
390 
391  strError = getMappingError(strKey, _contextMappingKeyArray[uiAncestorKey] +
392  " not set", pInstanceConfigurableElement);
393 
394  return false;
395  }
396  }
397 
398  // Then check configurable element size is correct
399  if (pInstanceConfigurableElement->getFootPrint() >
400  pSubsystemObjectCreator->getMaxConfigurableElementSize()) {
401 
402  string strSizeError = "Size should not exceed " +
403  std::to_string(
404  pSubsystemObjectCreator->getMaxConfigurableElementSize());
405 
406  strError = getMappingError(strKey, strSizeError, pInstanceConfigurableElement);
407 
408  return false;
409  }
410 
411  // Do create object and keep its track
412  _subsystemObjectList.push_back(pSubsystemObjectCreator->objectCreate(
413  *pStrValue, pInstanceConfigurableElement, context, _logger));
414 
415  // Indicate subsytem creation to caller
416  bHasCreatedSubsystemObject = true;
417 
418  // The subsystem Object has been instantiated, no need to continue looking for an
419  // instantiation mapping
420  break;
421  }
422  }
423 
424  return true;
425 }

427 // From IMapper
428 // Handle a configurable element mapping
429 bool CSubsystem::mapBegin(CInstanceConfigurableElement* pInstanceConfigurableElement,
430  bool& bKeepDiving, string& strError)
431 {
432  // Get current context
433  CMappingContext context = _contextStack.top();
434 
435  // Add mapping in context
436  if (!handleMappingContext(pInstanceConfigurableElement, context,
437  strError)) {
438 
439  return false;
440  }
441 
442  // Push context
443  _contextStack.push(context);
444 
445  // Assume diving by default
446  bKeepDiving = true;
447 
448  // Deal with ambiguous usage of parameter blocks
449  bool bShouldCreateSubsystemObject = true;
450 
451  switch(pInstanceConfigurableElement->getType()) {
452 
453  case CInstanceConfigurableElement::EComponent:
454  case CInstanceConfigurableElement::EParameterBlock:
455  // Subsystem object creation is optional in parameter blocks
456  bShouldCreateSubsystemObject = false;
457  // No break
458  case CInstanceConfigurableElement::EBitParameterBlock:
459  case CInstanceConfigurableElement::EParameter:
460  case CInstanceConfigurableElement::EStringParameter:
461 
462  bool bHasCreatedSubsystemObject;
463 
464  if (!handleSubsystemObjectCreation(pInstanceConfigurableElement, context,
465  bHasCreatedSubsystemObject, strError)) {
466 
467  return false;
468  }
469  // Check for creation error
470  if (bShouldCreateSubsystemObject && !bHasCreatedSubsystemObject) {
471 
472  strError = getMappingError("Not found",
473  "Subsystem object mapping key is missing",
474  pInstanceConfigurableElement);
475  return false;
476  }
477  // Not created and no error, keep diving
478  bKeepDiving = !bHasCreatedSubsystemObject;
479 
480  return true;
481 
482  default:
483  assert(0);
484  return false;
485  }
486 }
