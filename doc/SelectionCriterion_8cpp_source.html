63 void CSelectionCriterion::setCriterionState(int iState)
64 {
65  // Check for a change
66  if (_iState != iState) {
67 
68  _iState = iState;
69 
70  _logger.info() << "Selection criterion changed event: "
71  << getFormattedDescription(false, false);
72 
73  // Check if the previous criterion value has been taken into account (i.e. at least one Configuration was applied
