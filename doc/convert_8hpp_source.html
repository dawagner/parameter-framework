64 template<typename T>
65 static inline bool convertTo(const std::string &str, T &result)
66 {
67  /* Check that conversion to that type is allowed.
68  * If this fails, this means that this template was not intended to be used
69  * with this type, thus that the result is undefined. */
70  static_assert(ConvertionAllowed<T>::value, "convertTo does not support this conversion");
71 
72  if (str.find_first_of(std::string("\r\n\t\v ")) != std::string::npos) {
73  return false;
74  }
75 
76  /* Check for a '-' in string. If type is unsigned and a - is found, the
77  * parsing fails. This is made necessary because "-1" is read as 65535 for
78  * uint16_t, for example */
79  if (str.find("-") != std::string::npos
80  && !std::numeric_limits<T>::is_signed) {
81  return false;
82  }
83 
84  std::stringstream ss(str);
85 
86  /* Sadly, the stream conversion does not handle hexadecimal format, thus
87  * check is done manually */
88  if (str.substr(0, 2) == "0x") {
89  if (std::numeric_limits<T>::is_integer) {
90  ss >> std::hex >> result;
91  }
92  else {
93  /* Conversion undefined for non integers */
94  return false;
95  }
96  } else {
97  ss >> result;
98  }
99 
100  return ss.eof() && !ss.fail() && !ss.bad();
101 }

103 template<typename T, typename Via>
104 static inline bool convertToVia(const std::string &str, T &result)
105 {
106  /* Check that conversion to that type is allowed.
107  * If this fails, this means that this template was not intended to be used
108  * with this type, thus that the result is undefined. */
109  static_assert(ConvertionAllowedVia<T, Via>::value,
110  "convertToVia does not support this conversion");
111 
112  /* We want to override the behaviour of convertTo<T> with that of
113  * convertTo<Via> and then safely cast the result into a T. */
114  Via res;
115 
116  if (!convertTo<Via>(str, res)) {
117  return false;
118  }
119 
120  if ((res > std::numeric_limits<T>::max())
121  or (res < std::numeric_limits<T>::min())) {
122  return false;
123  }
124 
125  result = static_cast<T>(res);
126  return true;
127 }
