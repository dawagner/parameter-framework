113 bool CXmlDocSource::populate(CXmlSerializingContext& serializingContext)
114 {
115  // Check that the doc has been created
116  if (!_pDoc) {
117 
118  serializingContext.setError("Could not parse document ");
119 
120  return false;
121  }
122 
123  // Validate if necessary
124  if (_bValidateWithSchema)
125  {
126  if (!isInstanceDocumentValid()) {
127 
128  serializingContext.setError("Document is not valid");
129 
130  return false;
131  }
132  }
133 
134  // Check Root element type
135  if (getRootElementName() != _strRootElementType) {
136 
137  serializingContext.setError("Error: Wrong XML structure document ");
138  serializingContext.appendLineToError("Root Element " + getRootElementName()
139  + " mismatches expected type " + _strRootElementType);
140 
141  return false;
142  }
143 
144  if (!_strNameAttributeName.empty()) {
145 
146  string strRootElementNameCheck = getRootElementAttributeString(_strNameAttributeName);
147 
148  // Check Root element name attribute (if any)
149  if (!_strRootElementName.empty() && strRootElementNameCheck != _strRootElementName) {
150 
151  serializingContext.setError("Error: Wrong XML structure document ");
152  serializingContext.appendLineToError(_strRootElementType + " element "
153  + _strRootElementName + " mismatches expected "
154  + _strRootElementType + " type "
155  + strRootElementNameCheck);
156 
157  return false;
158  }
159  }
160 
161  return true;
162 }

164 bool CXmlDocSource::isInstanceDocumentValid()
165 {
166 #ifdef LIBXML_SCHEMAS_ENABLED
167  string schemaUri = getSchemaUri();
168 
169  xmlDocPtr pSchemaDoc = xmlReadFile(schemaUri.c_str(), NULL, XML_PARSE_NONET);
170 
171  if (!pSchemaDoc) {
172  // Unable to load Schema
173  return false;
174  }
175 
176  xmlSchemaParserCtxtPtr pParserCtxt = xmlSchemaNewDocParserCtxt(pSchemaDoc);
177 
178  if (!pParserCtxt) {
179 
180  // Unable to create schema context
181  xmlFreeDoc(pSchemaDoc);
182  return false;
183  }
184 
185  // Get Schema
186  xmlSchemaPtr pSchema = xmlSchemaParse(pParserCtxt);
187 
188  if (!pSchema) {
189 
190  // Invalid Schema
191  xmlSchemaFreeParserCtxt(pParserCtxt);
192  xmlFreeDoc(pSchemaDoc);
193  return false;
194  }
195  xmlSchemaValidCtxtPtr pValidationCtxt = xmlSchemaNewValidCtxt(pSchema);
196 
197  if (!pValidationCtxt) {
198 
199  // Unable to create validation context
200  xmlSchemaFree(pSchema);
201  xmlSchemaFreeParserCtxt(pParserCtxt);
202  xmlFreeDoc(pSchemaDoc);
203  return false;
204  }
205 
206  bool isDocValid = xmlSchemaValidateDoc(pValidationCtxt, _pDoc) == 0;
207 
208  xmlSchemaFreeValidCtxt(pValidationCtxt);
209  xmlSchemaFree(pSchema);
210  xmlSchemaFreeParserCtxt(pParserCtxt);
211  xmlFreeDoc(pSchemaDoc);
212 
213  return isDocValid;
214 #else
215  return true;
216 #endif
217 }

219 std::string CXmlDocSource::mkUri(const std::string& base, const std::string& relative)
220 {
221  xml_unique_ptr baseUri(xmlPathToURI((const xmlChar *)base.c_str()), xmlFree);
222  xml_unique_ptr relativeUri(xmlPathToURI((const xmlChar *)relative.c_str()), xmlFree);

224  xml_unique_ptr xmlUri(xmlBuildURI(relativeUri.get(), baseUri.get()), xmlFree);

232 _xmlDoc* CXmlDocSource::mkXmlDoc(const string& source, bool fromFile, bool xincludes, CXmlSerializingContext& serializingContext)
233 {
234  _xmlDoc* doc = NULL;
235  if (fromFile) {
236  doc = xmlReadFile(source.c_str(), NULL, 0);
237  } else {
238  doc = xmlReadMemory(source.c_str(), (int)source.size(), "", NULL, 0);
239  }
240 
241  if (doc == NULL) {
242  string errorMsg = "libxml failed to read";
243  if (fromFile) {
244  errorMsg += " \"" + source + "\"";
245  }
246  serializingContext.appendLineToError(errorMsg);
247 
248  return NULL;
249  }
250 
251  if (xincludes and (xmlXIncludeProcess(doc) < 0)) {
252  serializingContext.appendLineToError("libxml failed to resolve XIncludes");
253 
254  xmlFreeDoc(doc);
255  doc = NULL;
256  }
257 
258  return doc;
