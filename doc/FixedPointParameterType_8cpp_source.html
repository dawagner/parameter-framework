332 // Data conversion
333 int32_t CFixedPointParameterType::doubleToBinaryQnm(double dValue) const
334 {
335  // For Qn.m number, multiply by 2^n and round to the nearest integer
336  int32_t iData = static_cast<int32_t>(round(dValue * double(1UL << _uiFractional)));
337  // Left justify
338  // For a Qn.m number, shift 32 - (n + m + 1) bits to the left (the rest of
339  // the bits aren't used)
340  iData <<= getSize() * 8 - getUtilSizeInBits();
341 
342  return iData;
343 }

347 {
