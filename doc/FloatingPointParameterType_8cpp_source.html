87 bool CFloatingPointParameterType::fromXml(const CXmlElement& xmlElement,

89 {

93 
94  // Size support check: only floats are supported

96  if (sizeInBits != sizeof(float) * CHAR_BIT) {

100 
101  return false;
102  }
103 
104  setSize(sizeInBits / CHAR_BIT);
105 
106  xmlElement.getAttribute("Min", _fMin);
107  xmlElement.getAttribute("Max", _fMax);
108 
109  if (_fMin > _fMax) {

111  ") can't be greater than Max (" + std::to_string(_fMax) + ")");

118 bool CFloatingPointParameterType::toBlackboard(
119  const string& strValue,
120  uint32_t& uiValue,
121  CParameterAccessContext& parameterAccessContext) const
122 {
123  // Check Value integrity
124  if (utility::isHexadecimal(strValue) && !parameterAccessContext.valueSpaceIsRaw()) {
125 
126  parameterAccessContext.setError("Hexadecimal values are not supported for "
127  + getKind()
128  + " when selected value space is real: "
129  + strValue);
130 
131  return false;
132  }
133 
134  if (parameterAccessContext.valueSpaceIsRaw()) {
135  // Raw value: interpret the user input as the memory content of the
136  // parameter
137  if (!convertTo(strValue, uiValue)) {
138 
139  parameterAccessContext.setError("Value '" + strValue + "' is invalid");
140  return false;
141  }
142 
143  float fData = reinterpret_cast<const float &>(uiValue);
144 
145  // Check against NaN or infinity
146  if (!std::isfinite(fData)) {
147 
148  parameterAccessContext.setError("Value " + strValue + " is not a finite number");
149  return false;
150  }
151 
152  if (!checkValueAgainstRange(fData)) {
153 
154  setOutOfRangeError(strValue, parameterAccessContext);
155  return false;
156  }
157  return true;
158  }
159  else {
160 
161  float fValue = 0.0f;
162 
163  // Interpret the user input as float
164  if (!convertTo(strValue, fValue)) {
165 
166  parameterAccessContext.setError("Value " + strValue + " is invalid");
167  return false;
168  }
169 
170  if (!checkValueAgainstRange(fValue)) {
171 
172  setOutOfRangeError(strValue, parameterAccessContext);
173  return false;
174  }
175 
176  // Move to the "raw memory" value space
177  // Using an intermediary reference variable to avoid klocwork false positive
178  const uint32_t &value = reinterpret_cast<const uint32_t &>(fValue);
179  uiValue = value;
180  return true;
181  }
182 }
