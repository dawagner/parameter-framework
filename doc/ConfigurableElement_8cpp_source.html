50 bool CConfigurableElement::serializeXmlSettings(CXmlElement& xmlConfigurationSettingsElementContent, CConfigurationAccessContext& configurationAccessContext) const
51 {
52  size_t uiNbChildren = getNbChildren();
53 
54  if (!configurationAccessContext.serializeOut()) {
55  // Just do basic checks and propagate to children
56  CXmlElement::CChildIterator it(xmlConfigurationSettingsElementContent);
57 
58  CXmlElement xmlChildConfigurableElementSettingsElement;
59 
60  // Propagate to children
61  for (size_t index = 0; index < uiNbChildren; index++) {
62 
63  // Get child
64  const CConfigurableElement* pChildConfigurableElement = static_cast<const CConfigurableElement*>(getChild(index));
65 
66  if (!it.next(xmlChildConfigurableElementSettingsElement)) {
67 
68  // Structure error
69  configurationAccessContext.setError("Configuration settings parsing: Settings don't conform to structure of configurable element " + getName());
70 
71  return false;
72  }
73 
74  // Check element type matches in type
75  if (xmlChildConfigurableElementSettingsElement.getType() != pChildConfigurableElement->getKind()) {
76 
77  // Type error
78  configurationAccessContext.setError("Configuration settings parsing: Settings for configurable element " + pChildConfigurableElement->getName() + " does not match expected type: " + xmlChildConfigurableElementSettingsElement.getType() + " instead of " + pChildConfigurableElement->getKind());
79 
80  return false;
81  }
82 
83  // Check element type matches in name
84  if (xmlChildConfigurableElementSettingsElement.getNameAttribute() != pChildConfigurableElement->getName()) {
85 
86  // Name error
87  configurationAccessContext.setError("Configuration settings parsing: Under configurable elememnt " + getName() + ", expected element name " + pChildConfigurableElement->getName() + " but found " + xmlChildConfigurableElementSettingsElement.getNameAttribute() + " instead");
88 
89  return false;
90  }
91 
92  // Parse child configurable element's settings
93  if (!pChildConfigurableElement->serializeXmlSettings(xmlChildConfigurableElementSettingsElement, configurationAccessContext)) {
94 
95  return false;
96  }
97  }
98  // There should remain no configurable element to parse
99  if (it.next(xmlChildConfigurableElementSettingsElement)) {
100 
101  // Structure error
102  configurationAccessContext.setError("Configuration settings parsing: Settings don't conform to structure of configurable element " + getName());
103 
104  return false;
105  }
106  } else {
107  // Propagate to children
108  for (size_t index = 0; index < uiNbChildren; index++) {
109 
110  const CConfigurableElement* pChildConfigurableElement = static_cast<const CConfigurableElement*>(getChild(index));
111 
112  // Create corresponding child element
113  CXmlElement xmlChildConfigurableElementSettingsElement;
114 
115  xmlConfigurationSettingsElementContent.createChild(xmlChildConfigurableElementSettingsElement, pChildConfigurableElement->getKind());
116 
117  // Handle element name attribute
118  xmlChildConfigurableElementSettingsElement.setNameAttribute(pChildConfigurableElement->getName());
119 
120  // Propagate
121  pChildConfigurableElement->serializeXmlSettings(xmlChildConfigurableElementSettingsElement, configurationAccessContext);
122  }
123  }
124  // Done
125  return true;
126 }
