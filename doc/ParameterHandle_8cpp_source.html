236 bool CParameterHandle::checkGetValidity(bool asArray, string& error) const
237 {
238  if (asArray != isArray()) {
239 
240  auto toStr = [](bool array) { return array ? "an array" : "a scalar"; };
241  error = "Can not get \"" + getPath() + "\" as " + toStr(asArray) +
242  " because it is " + toStr(isArray());
243  return false;
244  }
245 
246  return true;
247 }

249 // Access validity
250 bool CParameterHandle::checkSetValidity(size_t arrayLength, string& error) const
251 {
252  // Settings a parameter necessitates the right to get it
253  if (not checkGetValidity(arrayLength != 0, error)) {
254  return false;
255  }
256 
257  if (!isRogue()) {
258 
259  error = "Can not set parameter \"" + getPath() + "\" as it is not rogue.";
260  return false;
261  }
262 
263  if (arrayLength && (arrayLength != getArrayLength())) {
264 
265  using std::to_string;
266  error = "Array length mismatch for \"" + getPath() +
267  "\", expected: " + to_string(getArrayLength()) +
268  ", got: " + to_string(arrayLength);
269  return false;
270  }
271 
272  return true;
