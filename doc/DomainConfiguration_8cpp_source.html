65 // XML configuration settings parsing
66 bool CDomainConfiguration::parseSettings(CXmlElement& xmlConfigurationSettingsElement, CXmlDomainImportContext& context)
67 {
68  // Parse configurable element's configuration settings
69  CXmlElement::CChildIterator it(xmlConfigurationSettingsElement);
70 
71  CXmlElement xmlConfigurableElementSettingsElement;
72  auto insertLocation = begin(mAreaConfigurationList);
73 
74  while (it.next(xmlConfigurableElementSettingsElement)) {
75 
76  // Retrieve area configuration
77  string configurableElementPath;
78  xmlConfigurableElementSettingsElement.getAttribute("Path", configurableElementPath);
79 
80  auto areaConfiguration = findAreaConfigurationByPath(configurableElementPath);
81  if (areaConfiguration == end(mAreaConfigurationList)) {
82 
83  context.setError("Configurable Element " + configurableElementPath + " referred to by Configuration " + getPath() + " not associated to Domain");
84 
85  return false;
86  }
87  // Parse
88  if (!importOneConfigurableElementSettings(areaConfiguration->get(), xmlConfigurableElementSettingsElement, context)) {
89 
90  return false;
91  }
92  // Take into account the new configuration order by moving the configuration associated to
93  // the element to the n-th position of the configuration list.
94  // It will result in prepending to the configuration list wit the configuration of all
95  // elements found in XML, keeping the order of the processing of the XML file.
96  mAreaConfigurationList.splice(insertLocation, mAreaConfigurationList, areaConfiguration);
97  // areaConfiguration is still valid, but now refer to the reorderer list
98  insertLocation = std::next(areaConfiguration);
99  }
100  return true;
101 }

125 // Serialize one configuration for one configurable element
126 bool CDomainConfiguration::importOneConfigurableElementSettings(
127  CAreaConfiguration *areaConfiguration,
128  CXmlElement& xmlConfigurableElementSettingsElement,
129  CXmlDomainImportContext& context)
130 {
131  const CConfigurableElement* destination = areaConfiguration->getConfigurableElement();
132 
133  // Check structure
134  if (xmlConfigurableElementSettingsElement.getNbChildElements() != 1) {
135 
136  // Structure error
137  context.setError(
138  "Struture error encountered while parsing settings of " +
139  destination->getKind() + " " + destination->getName() +
140  " in Configuration " + getPath());
141 
142  return false;
143  }
144 
145  // Element content
146  CXmlElement xmlConfigurableElementSettingsElementContent;
147  // Check name and kind
148  if (!xmlConfigurableElementSettingsElement.getChildElement(
149  destination->getKind(),
150  destination->getName(),
151  xmlConfigurableElementSettingsElementContent)) {
152 
153  context.setError(
154  "Couldn't find settings for " +
155  destination->getKind() + " " + destination->getName() +
156  " for Configuration " + getPath());
157 
158  return false;
159  }
160 
161  // Create configuration access context
162  string error;
163  CConfigurationAccessContext configurationAccessContext(error, false);
164 
165  // Have domain configuration parse settings for configurable element
166  bool success = areaConfiguration->serializeXmlSettings(
167  xmlConfigurableElementSettingsElementContent,
168  configurationAccessContext);
169 
170  context.appendToError(error);
171  return success;
172 }

213 bool CDomainConfiguration::setElementSequence(const std::vector<string> &newElementSequence, string &error)
214 {
215  std::vector<string> elementSequenceSet;
216  auto insertLocation = begin(mAreaConfigurationList);
217 
218  for (const std::string &elementPath : newElementSequence) {
219 
220  auto areaConfiguration = findAreaConfigurationByPath(elementPath);
221  if (areaConfiguration == end(mAreaConfigurationList)) {
222 
223  error = "Element " + elementPath + " not found in domain";
224 
225  return false;
226  }
227  auto it = find(begin(elementSequenceSet), end(elementSequenceSet), elementPath);
228  if (it != end(elementSequenceSet)) {
229  error = "Element " + elementPath + " provided more than once";
230  return false;
231  }
232  elementSequenceSet.push_back(elementPath);
233  // Take into account the new configuration order by moving the configuration associated to
234  // the element to the n-th position of the configuration list.
235  // It will result in prepending to the configuration list wit the configuration of all
236  // elements found in XML, keeping the order of the processing of the XML file.
237  mAreaConfigurationList.splice(insertLocation, mAreaConfigurationList, areaConfiguration);
238  // areaConfiguration is still valid, but now refer to the reorderer list
239  insertLocation = std::next(areaConfiguration);
240  }
241  return true;
242 }
