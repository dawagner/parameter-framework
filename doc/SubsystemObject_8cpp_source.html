111 bool CSubsystemObject::sync(CParameterBlackboard& parameterBlackboard, bool bBack, string& strError)
112 {
113  // Get blackboard location
114  _blackboard = &parameterBlackboard;
115  // Access index init
116  _accessedIndex = 0;
117 
118 #ifdef SIMULATION
119  return true;
120 #endif
121 
122  // Retrieve subsystem
123  const CSubsystem* pSubsystem = _pInstanceConfigurableElement->getBelongingSubsystem();
124 
125  // Get it's health insdicator
126  bool bIsSubsystemAlive = pSubsystem->isAlive();
127 
128  // Check subsystem health
129  if (!bIsSubsystemAlive) {
130 
131  strError = "Susbsystem not alive";
132  }
133 
134  // Synchronize to/from HW
135  if (!bIsSubsystemAlive || !accessHW(bBack, strError)) {
136 
137  // Fall back to parameter default initialization
138  if (bBack) {
139 
140  setDefaultValues(parameterBlackboard);
141  }
142  return false;
143  }
144 
145  return true;
146 }
